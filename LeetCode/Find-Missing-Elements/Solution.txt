class Solution {
    public List<Integer> findMissingElements(int[] nums) {
        Arrays.sort(nums);
        List<Integer> missing = new ArrayList<>();
        int start=nums[0];
        int end=nums[nums.length-1];
        for(int i=start;i <=end;i++){
            if(Arrays.binarySearch(nums,i)<0){
                missing.add(i);
            }
        }
        return missing;
        
    }
}
